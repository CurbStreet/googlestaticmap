  def initialize(attrs={})
    defaults = {:width => 500, :height => 350, :markers => [],
                :sensor => false, :maptype => "roadmap", :paths => []}
    attributes = defaults.merge(attrs)
    attributes.each {|k,v| self.send("#{k}=".to_sym,v)}
  end
